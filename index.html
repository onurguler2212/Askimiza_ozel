  <script>

    const bgm = document.getElementById("bgm");

    const note = document.getElementById("note");

    const playBtn = document.getElementById("playBtn");

    let playing = false;

    function toggleMusic(){

      if(!playing){

        bgm.muted = false;

        bgm.volume = 1.0;

        bgm.play().then(()=>{

          playing = true;

          playBtn.textContent = "⏸ Müziği Durdur";

        }).catch(()=>{

          alert("🔊 Müzik başlamadı. Lütfen butona bir kez daha dokun 💖");

        });

      }else{

        bgm.pause();

        playing = false;

        playBtn.textContent = "🎵 Müziği Başlat";

      }

    }

    function toggleNote(){

      note.style.display = (note.style.display === "block") ? "none" : "block";

      spawnHearts(10);

    }

    function spawnHearts(count){

      for(let i=0;i<count;i++){

        const h=document.createElement("div");

        h.className="heart";

        h.textContent=Math.random()>0.25?"❤":"🤍";

        h.style.left=Math.random()*100+"vw";

        h.style.animationDuration=(Math.random()*3+4)+"s";

        document.body.appendChild(h);

        setTimeout(()=>h.remove(),7000);

      }

    }

    window.addEventListener("load",()=>{

      spawnHearts(6);

    });

  </script>
